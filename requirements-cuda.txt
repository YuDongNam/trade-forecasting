# CUDA-enabled PyTorch installation
# Use this if you have NVIDIA GPU with CUDA support
# Install this AFTER installing base requirements:
#   pip install -r requirements.txt
#   pip install -r requirements-cuda.txt

# Choose the appropriate version based on your CUDA installation

# torchaudio>=2.0.0,<2.5.0 --index-url https://download.pytorch.org/whl/cu118

# For CUDA 12.1:
# torch>=2.0.0,<2.5.0 --index-url https://download.pytorch.org/whl/cu121
